// server.js import express from "express"; import Stripe from "stripe"; import cors from "cors"; const app = express(); app.use(cors()); app.use(express.json()); const stripe = new Stripe(process.env.STRIPE_SECRET_KEY); // sk_test_... // calculează suma (în pence) după durată function pricePence(hoursStr) { const h = parseFloat(hoursStr); if (isNaN(h)) throw new Error("Invalid duration"); if (h === 0.5) return 200; // £2.00 return Math.round(h * 400); // £4.00 / oră } app.post("/create-checkout", async (req, res) => { try { const { plate, loc, h } = req.body; const amount = pricePence(h); const session = await stripe.checkout.sessions.create({ mode: "payment", currency: "gbp", line_items: [ { price_data: { currency: "gbp", product_data: { name: `Parking ${plate} @ ${loc} (${h}h)` }, unit_amount: amount, }, quantity: 1, }, ], // URLs după plată success_url: "https://qrtap.org/success.html", cancel_url: "https://qrtap.org/cancel.html", // Apple Pay / Google Pay on by default payment_method_types: ["card"], allow_promotion_codes: false, billing_address_collection: "auto", }); res.json({ url: session.url }); } catch (err) { console.error(err); res.status(400).json({ error: err.message }); } }); app.listen(4242, () => console.log("Server on http://localhost:4242")); STRIPE_SECRET_KEY=sk_test_xxxxxxxxxxxxxxxxxxxxxxxxx
